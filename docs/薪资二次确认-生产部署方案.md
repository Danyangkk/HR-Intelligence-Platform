### 步骤1: 清理调试日志

修改 `frontend/index.html` 的 `selectL3()` 函数，移除所有 `console.log`：

```javascript
async function selectL3(id){
  hideBackFloat();
  
  // 检查是否是薪资L3
  const isPayroll = typeof isPayrollL3==='function' && isPayrollL3(id);
  
  if(isPayroll){
    // 业务超管需要二次确认
    if(AUTH.role === 'biz_super_admin'){
      if(typeof window.ensurePayrollAccess !== 'function'){
        toast('系统错误：缺少权限确认函数', 'danger');
        return;
      }
      const ok = await window.ensurePayrollAccess({
        target_ref:'数据中台查询', 
        entry: typeof l3Name==='function' ? l3Name(id) : '薪资数据'
      });
      if(!ok) return;
    } else {
      // 非业务超管直接拦截
      toast('无权访问薪资数据', 'danger');
      return;
    }
  }
  
  state.activeL3=id; 
  resetView(); 
  void renderAll();
}
```


```nginx

# JS: 1小时缓存
```

### 2. 长期改进建议

**最佳实践**：引入构建工具（Webpack/Vite），为JS/CSS文件名添加hash：
```
app.abc123.js  ← 内容变化时hash自动改变
```

这样可以：
- JS/CSS文件永久缓存（1年）
- 代码更新时自动失效旧缓存
- 无需清理浏览器缓存
