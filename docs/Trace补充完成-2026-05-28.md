```python
# 新增: top_score 字段（RAG相似度最高分）
top_score = 0.0
for block in merged_evidence:
    if block.get("kind") == "documents":
        for hit in block.get("hits") or []:
            if isinstance(hit, dict) and "score" in hit:
                try:
                    score = float(hit.get("score") or 0)
                    top_score = max(top_score, score)
                except (ValueError, TypeError):
                    pass
if top_score > 0:
    decision["top_score"] = round(top_score, 3)
```

```sql
-- 1. 检查badcase分布
SELECT 
    badcase_reason,
    COUNT(*) as count,
    AVG(total_ms) as avg_ms,
    AVG(replan_count) as avg_replan
FROM agent_run
WHERE auto_badcase = true
GROUP BY badcase_reason
ORDER BY count DESC;

-- 2. 检查decision字段完整性
SELECT 
    node,
    COUNT(*) as total,
    COUNT(CASE WHEN decision IS NOT NULL THEN 1 END) as has_decision,
    COUNT(CASE WHEN decision IS NULL THEN 1 END) as no_decision
FROM agent_node_trace
GROUP BY node
ORDER BY node;

-- 3. 检查RAG 0命中场景
SELECT 
    r.id,
    r.intent,
    r.outcome,
    r.auto_badcase,
    r.badcase_reason,
    t.decision->>'chunks_hit' as chunks_hit
FROM agent_run r
JOIN agent_node_trace t ON r.id = t.run_id
WHERE t.node = 'document'
  AND (t.decision->>'chunks_hit')::int = 0
LIMIT 10;
```
